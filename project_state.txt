﻿using System.Text.Json.Serialization;

namespace CleaningCRM.API.Models
{
    public class Order
    {
        public int Id { get; set; }
        public int ClientId { get; set; }
        public int ServiceId { get; set; }
        public int EmployeeId { get; set; }
        public DateTime OrderDate { get; set; } = DateTime.Now;
        public string Address { get; set; }
        public string Status { get; set; } = "Новый";

        // Добавьте [JsonIgnore], чтобы не требовать эти поля в JSON-запросе
        [JsonIgnore]
        public Client? Client { get; set; }
        [JsonIgnore]
        public Service? Service { get; set; }
        [JsonIgnore]
        public Employee? Employee { get; set; }
    }
}
## Program.cs:
using Microsoft.AspNetCore.Authentication.JwtBearer;
using Microsoft.IdentityModel.Tokens;
using System.Text;
using Microsoft.EntityFrameworkCore;
using CleaningCRM.API.Data;

var builder = WebApplication.CreateBuilder(args);

// CORS
builder.Services.AddCors(options =>
{
    options.AddPolicy("AllowFrontend", policy =>
    {
        policy.AllowAnyOrigin()
              .AllowAnyHeader()
              .AllowAnyMethod();
    });
});

// JWT Authentication
builder.Services.AddAuthentication(JwtBearerDefaults.AuthenticationScheme)
    .AddJwtBearer(options =>
    {
        options.TokenValidationParameters = new TokenValidationParameters
        {
            ValidateIssuer = true,
            ValidateAudience = true,
            ValidateLifetime = true,
            ValidateIssuerSigningKey = true,
            ValidIssuer = "CleaningCRM",
            ValidAudience = "CleaningCRM",
            IssuerSigningKey = new SymmetricSecurityKey(
                Encoding.UTF8.GetBytes("supersecretkeywithatleast32characterslong!"))
        };
    });

builder.Services.AddControllers();
builder.Services.AddDbContext<AppDbContext>(options =>
    options.UseSqlServer(builder.Configuration.GetConnectionString("DefaultConnection")));

var app = builder.Build();

app.UseCors("AllowFrontend");
app.UseAuthentication();
app.UseAuthorization();
app.MapControllers();

app.Run();

        const API_URL = 'http://localhost:5000/api';
        let currentToken = null;
        let currentRole = null;

        // === АВТОРИЗАЦИЯ ===
        async function login() {
            const username = document.getElementById('username').value;
            const password = document.getElementById('password').value;

            if (!username || !password) {
                document.getElementById('loginError').innerText = 'Введите логин и пароль';
                return;
            }

            try {
                const res = await fetch(`${API_URL}/auth/login`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ username, password })
                });

                if (!res.ok) throw new Error('Ошибка входа');

                const data = await res.json();
                currentToken = data.token;
                currentRole = data.role;

                document.getElementById('userName').innerHTML = data.username;
                document.getElementById('userRole').innerHTML = data.role;
                document.getElementById('loginPanel').style.display = 'none';
                document.getElementById('mainPanel').style.display = 'block';

                loadInterfaceByRole();
            } catch(e) {
                document.getElementById('loginError').innerText = 'Неверный логин или пароль';
            }
        }

        function logout() {
            currentToken = null;
            currentRole = null;
            document.getElementById('loginPanel').style.display = 'block';
            document.getElementById('mainPanel').style.display = 'none';
            document.getElementById('username').value = '';
            document.getElementById('password').value = '';

        // === АДМИНИСТРАТОР: КЛИЕНТЫ ===
        async function loadClients() {
            const res = await fetch(`${API_URL}/clients`);
            const clients = await res.json();
            const tbody = document.querySelector('#clientsTable tbody');
            if (!tbody) return;
            tbody.innerHTML = '';
            clients.forEach(c => {
                tbody.innerHTML += `<tr>
                    <td>${c.id}</td>
                    <td>${c.name}</td>
                    <td>${c.phone}</td>
                    <td>${c.email}</td>
                    <td><button class="delete" onclick="deleteClient(${c.id})">Удалить</button></td>
                </tr>`;
            });

        window.addClient = async () => {
            const name = document.getElementById('cName')?.value;
            const phone = document.getElementById('cPhone')?.value;
            const email = document.getElementById('cEmail')?.value;
            if (!name || !phone || !email) {
                alert('Заполните все поля');
                return;
            }
            await fetch(`${API_URL}/clients`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ name, phone, email })
            });
            document.getElementById('cName').value = '';
            document.getElementById('cPhone').value = '';
            document.getElementById('cEmail').value = '';
            await loadClients();
            await loadSelects();
        };

        window.deleteClient = async (id) => {
            if (confirm('Удалить клиента?')) {
                await fetch(`${API_URL}/clients/${id}`, { method: 'DELETE' });
                await loadClients();
                await loadSelects();
            }
        };

        // === АДМИНИСТРАТОР: ЗАКАЗЫ ===
        async function loadOrders() {
            const res = await fetch(`${API_URL}/orders`);
            const orders = await res.json();
            const tbody = document.querySelector('#ordersTable tbody');
            if (!tbody) return;
            tbody.innerHTML = '';
            orders.forEach(o => {
                tbody.innerHTML += `<tr>
                    <td>${o.id}</td>
                    <td>${o.client?.name || 'Нет'}</td>
                    <td>${o.service?.name || 'Нет'}</td>
                    <td>${o.employee?.fullName || 'Нет'}</td>
                    <td>${o.address}</td>
                    <td>
                        <select onchange="updateOrderStatus(${o.id}, this.value)">
                            <option ${o.status === 'Новый' ? 'selected' : ''}>Новый</option>
                            <option ${o.status === 'В работе' ? 'selected' : ''}>В работе</option>
                            <option ${o.status === 'Выполнен' ? 'selected' : ''}>Выполнен</option>
                            <option ${o.status === 'Оплачен' ? 'selected' : ''}>Оплачен</option>
                        </select>

                    <td><button class="delete" onclick="deleteOrder(${o.id})">Удалить</button></td>
                </tr>`;

        }

        async function loadSelects() {
            try {
                const [clients, services, employees] = await Promise.all([
                    fetch(`${API_URL}/clients`).then(r => r.json()),
                    fetch(`${API_URL}/services`).then(r => r.json()),
                    fetch(`${API_URL}/employees`).then(r => r.json())
                ]);

                const clientSelect = document.getElementById('oClient');
                const serviceSelect = document.getElementById('oService');
                const employeeSelect = document.getElementById('oEmployee');

                if (clientSelect) {
                    clientSelect.innerHTML = '<option value="">-- Выберите клиента --</option>';
                    clients.forEach(c => clientSelect.innerHTML += `<option value="${c.id}">${c.name}</option>`);
                }
                if (serviceSelect) {
                    serviceSelect.innerHTML = '<option value="">-- Выберите услугу --</option>';
                    services.forEach(s => serviceSelect.innerHTML += `<option value="${s.id}">${s.name} (${s.price} руб.)</option>`);
                }
                if (employeeSelect) {
                    employeeSelect.innerHTML = '<option value="">-- Выберите сотрудника --</option>';
                    employees.forEach(e => employeeSelect.innerHTML += `<option value="${e.id}">${e.fullName} (${e.position})</option>`);
                }
            } catch(e) { console.error(e); }
        }

      window.addOrder = async () => {
    const address = document.getElementById('oAddress')?.value;
    const clientId = document.getElementById('oClient')?.value;
    const serviceId = document.getElementById('oService')?.value;
    const employeeId = document.getElementById('oEmployee')?.value;

    if (!address || !clientId || !serviceId || !employeeId) {
        alert('Заполните все поля!');
        return;
    }

    const orderData = {
        clientId: parseInt(clientId),
        serviceId: parseInt(serviceId),
        employeeId: parseInt(employeeId),
        address: address,
        status: "Новый",
        orderDate: new Date().toISOString()
    };

    try {
        const response = await fetch(`${API_URL}/orders`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(orderData)
        });

        if (response.ok) {
            alert('Заказ создан!');
            document.getElementById('oAddress').value = '';
            document.getElementById('oClient').value = '';
            document.getElementById('oService').value = '';
            document.getElementById('oEmployee').value = '';
            await loadOrders();
        } else {
            const error = await response.text();
            console.error('Ошибка:', error);
            alert('Ошибка при создании заказа. Проверьте консоль (F12)');
        }
    } catch (error) {
        console.error('Ошибка:', error);
        alert('Не удалось создать заказ');
    }
};

            });
        }

        window.completeOrder = async (id) => {
            await fetch(`${API_URL}/orders/${id}`, {
                method: 'PUT',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify('Выполнен')
            });
            await loadMyOrders();
        };

        // === БУХГАЛТЕР ===
        async function loadPayments() {
            const res = await fetch(`${API_URL}/orders`);
            const orders = await res.json();
            const tbody = document.querySelector('#paymentsTable tbody');
            if (!tbody) return;
            tbody.innerHTML = '';
            orders.forEach(o => {
                tbody.innerHTML += `<tr>
                    <td>${o.id}</td>
                    <td>${o.client?.name || 'Нет'}</td>
                    <td>${o.service?.price || 0} руб.</td>
                    <td>${o.status}</td>
                    <td>${o.status !== 'Оплачен' ? `<button class="success" onclick="markPaid(${o.id})">💰 Отметить оплату</button>` : '💳 Оплачено'}</td>
                </tr>`;
            });

        window.markPaid = async (id) => {
            await fetch(`${API_URL}/orders/${id}`, {
                method: 'PUT',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify('Оплачен')
            });
            await loadPayments();
        };

        // === АДМИНИСТРАЦИЯ ===
        async function loadStats() {
            const res = await fetch(`${API_URL}/orders`);
            const orders = await res.json();
            const total = orders.length;
            const completed = orders.filter(o => o.status === 'Выполнен' || o.status === 'Оплачен').length;
            const totalSum = orders.reduce((sum, o) => sum + (o.service?.price || 0), 0);
            document.getElementById('statsContainer').innerHTML = `
                <div style="display: grid; grid-template-columns: repeat(3, 1fr); gap: 20px;">
                    <div style="background: #f0f2f5; padding: 20px; border-radius: 10px; text-align: center;">

                        <p style="font-size: 32px;">${total}</p>

                        <h3>✅ Выполнено</h3>
                        <p style="font-size: 32px;">${completed}</p>
                    </div>
                    <div style="background: #f0f2f5; padding: 20px; border-radius: 10px; text-align: center;">
                        <h3>💰 Выручка</h3>
                        <p style="font-size: 32px;">${totalSum} руб.</p>
                    </div>
                </div>
            `;
        }
